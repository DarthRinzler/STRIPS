(InferAdjacency (a b)
	(Pre (a.adjacent b))
	(Post (b.adjacent a))
)

(Move (obj src dest)
	(Pre 
		(and 
			(src.type location)
			(dest.type location)
			(obj.at src)
			(src.adjacent dest)
		)
	)
	(Post
		(and
			(obj.at dest)
			(not (obj.at src))
		)
	)
)

(Get (owner obj loc)
	(Pre
		(and
			(loc.type location)
			(owner.at loc)	
			(obj.at loc)
		)
	)
	(Post
		(owner.has obj)	
	)
)
