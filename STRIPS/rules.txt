(Eat (obj food)
	(Pre
		(obj has food)	
	)
	(Post
		(And
			(not (obj has food))	
			(not (obj IsHungry))
		)
	)
)

(InferAdjacency (a b)
	(Pre (a adjacent b))
	(Post (b adjacent a))
)

(Move (obj src dest)
	(Pre 
		(and 
			(src type location)
			(dest type location)
			(obj at src)
			(src adjacent dest)
		)
	)
	(Post
		(and
			(obj at dest)
			(not (obj at src))
		)
	)
)

(Transfer (src dest obj)
	(Pre
		(and
			(src has obj)	
			(not (dest has obj))
		)
	)
	(Post
		(and
			(not (src has obj))
			(dest has obj)
		)
	)
)


(Copy (src dest data)
	(Pre
		(src has data)	
	)
	(Post
		(and
			(src has data)
			(dest has data)
		)
	)
)

(Goal (player)
	(Pre
		(And
			(player.name.first.graham)
			(player.type.human)
			(player.at.home)	
			(not (player isHungry))
		)	
	)
	(Post
		(player wins)	
	)
)
