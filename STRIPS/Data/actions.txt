
(Work (person)
	(pre
		(person has energy)
	)
	(post
		(person has money)
		(not (person has energy))
	)
)

(Eat (obj, food)
	(Pre
		(food is edible)
		(obj has food)	
	)
	(Post
		(And
			(not (obj has food))	
			(not (obj IsHungry))
		)
	)
)

(Cook (obj)
	(Pre
		(obj is organic)	
	)
	(Post
		(obj is edible)	
	)
)

(InferAdjacency (a b)
	(Pre (a adjacent b))
	(Post (b adjacent a))
)

(Move (obj src dest)
	(Pre 
		(and 
			(src is location)
			(dest is location)
			(obj at src)
			(src adjacent dest)
		)
	)
	(Post
		(and
			(obj at dest)
			(not (obj at src))
		)
	)
)

(Buy (buyer seller obj loc)
	(Pre
		(and
			(buyer at loc)
			(seller at loc)
			(buyer has money)
			(seller has obj)	
		)
	)
	(Post
		(and
			(not (buyer has money))
			(not (seller has obj))
			(buyer has obj)
			(seller has money)
		)
	)
)

(Goal (player petitioner)

			(player is human)
			(player at home)	
			(not (player isHungry))
		)	
	)
	(Post
		(petitioner IsSatisfied)	
	)
)
